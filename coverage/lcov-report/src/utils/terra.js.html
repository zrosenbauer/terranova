/**

 *   stylesModifiers: Array // functions that will modify and override base configured styles
 * }
 * This allows dev to add differing featureTypes per map instance

// Methods

/**
 * Check if the map exists
 * @param {String} id
 * @returns {Boolean}
 */
export function terraExists(id) {
  return Boolean(terra[id]);
}
 
// Getters

/**
 * @param {String} id
 * @returns {Maps|null}
 */
export function getTerraById(id) {
  if (!terraExists(id)) {
    return null;
  }
 
  return terra[id];
}
 
/**
 * @param {String} id
 * @returns {Maps|null}
 */
export function getMapById(id) {
  if (!terraExists(id)) {
    return null;
  }
 
  return terra[id].map;
}
 
/**
 * @param {String} id
 * @returns {Data|null}
 */
export function getDataById(id) {
  if (!terraExists(id)) {
    return null;

  return terra[id].map.data;
}
